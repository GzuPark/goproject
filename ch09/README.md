# Chapter 09 - if문

```go
if 조건문 {
    문장                // 조건문이 true일 때 수행
}
```

```go
if 조건문1 {
    문장                // 조건문이 true일 때 수행
} else {
    문장                // 조건문이 false일 때 수행
}
```

```go
if 조건문 1 {
    문장                // 조건문1이 true일 때 수행
} else if 조건문2 {
    문장                // 조건문2가 true일 때 수행
} else if 조건문3 {
    문장                // 조건문3이 true일 때 수행
} else {                // else는 항상 맨 마지막에 위치, 생략 가능
    문장                // 모든 조건이 false일 때 수행
}
```

```go
if 초기문; 조건문 {     // 초기문을 먼저 실행하고 조건 검사 실행
    문장                // 조건문이 true일 때 수행
}
```
